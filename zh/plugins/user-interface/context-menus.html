export default class ExamplePlugin extends Plugin {

      const menu = new Menu();

      menu.addItem((item) =>

            new Notice("Copied");

        item
          .setTitle("Paste")
          .setIcon("paste")
          .onClick(() => {
            new Notice("Pasted");
          })
      );

      menu.showAtMouseEvent(event);
    });
  }
}
